// Basic site JS: nav toggle and dynamic year insertion
document.addEventListener('DOMContentLoaded', function () {
  // set year in multiple pages
  const year = new Date().getFullYear();
  document.querySelectorAll('#year, #yearAbout, #yearServices, #yearPortfolio, #yearContact').forEach(el => {
    if (el) el.textContent = year;
  });

  // nav toggles (works for multiple header instances)
  document.querySelectorAll('.nav-toggle').forEach(btn => {
    btn.addEventListener('click', function () {
      // find nav controlled by this button
      const targetId = btn.getAttribute('aria-controls') || btn.nextElementSibling?.id;
      const nav = document.getElementById(targetId) || btn.nextElementSibling;
      if (!nav) return;
      const expanded = btn.getAttribute('aria-expanded') === 'true';
      btn.setAttribute('aria-expanded', String(!expanded));
      nav.classList.toggle('mobile-open');
    });
  });

  // Improve accessibility: close mobile nav when clicking a nav link
  document.querySelectorAll('.site-nav a').forEach(a => {
    a.addEventListener('click', () => {
      const nav = a.closest('.site-nav');
      if (nav && nav.classList.contains('mobile-open')) {
        nav.classList.remove('mobile-open');
        const toggle = nav.parentElement.querySelector('.nav-toggle');
        if (toggle) toggle.setAttribute('aria-expanded', 'false');
      }
    });
  });
});
# Your Company — Static Website

This repository contains a simple 5-page static website (HTML/CSS/JS) ready to be hosted on GitHub Pages.

## How to use
1. Replace placeholder images in `assets/images/`.
2. Update copy (company name, emails, pricing) in the HTML files.
3. If you want contact form submissions, update the `form` action in `contact.html` (e.g., Formspree or other service).
4. Commit and push to a GitHub repo, then enable GitHub Pages in repository Settings (branch: `main` or `gh-pages`, folder: root).

## Deploy to GitHub Pages
- Create a new repository, push the files.
- In GitHub repo => Settings => Pages => select branch and root, Save.
- Your site will be published at `https://<username>.github.io/<repo>/`.
